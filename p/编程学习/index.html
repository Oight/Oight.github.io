long long fac(int n) {
    if (n == 1) {
        return 1;
    }

    return fac(n-1) * n;
}

fn main() {
    let mut stdin = String::new();

    io::stdin().read_line(&mut stdin).expect("");

        .trim()
        .parse()

fn factorial(n: i64) -> i64 {
    if n == 1 {
        return 1;
    }

    return n * factorial(n - 1);
}
